Algorithm twoSum(n, arr, target) {  
    hashmap = new HashMap<Integer, Integer>();  
 
    for (i = 0; i < n; i++) {  
        num = arr[i];  
        moreNeeded = target - num;  
 
        if (hashmap contains moreNeeded) {  
            return (hashmap[moreNeeded], i);  
        }  
 
        hashmap[num] = i;  
    }  
 
    return (-1, -1);  
}
